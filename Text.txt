//using namespace std;

int main()
{
    int n1, n2, max;

    n1 = 5;
    n2 = n1+5;

    // maximum value between n1 and n2 is stored in max
    max = (n1 > n2) ? n1 : n2;

    int a = 0;
    while(true){
        if (max % 2 == 1)
        {
            //cout << "LCM = " << max;
            a = a +2;
        }
        else
            a = a+1;
    }

    return a;
}
